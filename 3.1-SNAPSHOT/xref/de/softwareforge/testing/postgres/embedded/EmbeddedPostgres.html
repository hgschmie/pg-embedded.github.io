294     private void initDatabase() throws IOException {
295         ImmutableList.Builder<String> commandBuilder = ImmutableList.builder();
296         commandBuilder.add(pgBin("initdb"))
297                 .addAll(createLocaleOptions())
298                 .add("-A", "trust",
299                         "-U", PG_DEFAULT_USER,
300                         "-D", this.dataDirectory.getPath(),
301                         "-E", "UTF-8");
302         final Stopwatch watch = system(commandBuilder.build(), true);
303         logger.debug(format("initdb completed in %s", formatDuration(watch.elapsed())));
304     }

306     private void startDatabase() throws IOException {
307         checkState(!started.getAndSet(true), "database already started!");
308 
309         final ImmutableList.Builder<String> commandBuilder = ImmutableList.builder();
310         commandBuilder.add(pgBin("pg_ctl"),
311                 "-D", this.dataDirectory.getPath(),
312                 "-o", createInitOptions().stream().collect(Collectors.joining(" ")),
313                 "start"
314         );
315 
316         final Stopwatch watch = Stopwatch.createStarted();
317         final Process postmaster = spawn("pg", commandBuilder.build(), true);
318 
319         logger.info(format("started as pid %d on port %d", postmaster.pid(), port));
320         logger.debug(format("Waiting up to %s for server startup to finish", formatDuration(serverStartupWait)));
321 
322         Runtime.getRuntime().addShutdownHook(newCloserThread());
323 
324         checkState(waitForServerStartup(), "Could not start pg, interrupted?");
325         logger.debug(format("startup complete in %s", formatDuration(watch.elapsed())));
326     }

329         final ImmutableList.Builder<String> commandBuilder = ImmutableList.builder();
330         commandBuilder.add(pgBin("pg_ctl"),
331                 "-D", dataDirectory.getPath(),
332                 "stop",
333                 "-m", PG_STOP_MODE,
334                 "-t", PG_STOP_WAIT_S, "-w");
335 
336         final Stopwatch watch = system(commandBuilder.build(), true);
337         logger.debug(format("shutdown complete in %s", formatDuration(watch.elapsed())));
338     }

367     private boolean waitForServerStartup() throws IOException {
368         Throwable lastCause = null;
369         final long start = System.nanoTime();
370         final long maxWaitNs = TimeUnit.NANOSECONDS.convert(serverStartupWait.toMillis(), TimeUnit.MILLISECONDS);
371         while (System.nanoTime() - start < maxWaitNs) {
372             try {
373                 if (verifyReady()) {
374                     return true;
375                 }
376             } catch (final SQLException e) {
377                 lastCause = e;
378                 logger.trace("while waiting for server startup:", e);
379             }
380 
381             try {
382                 Thread.sleep(100);
383             } catch (final InterruptedException e) {
384                 Thread.currentThread().interrupt();
385                 return false;
386             }
387         }
388         throw new IOException("Gave up waiting for server to start after " + serverStartupWait.toMillis() + "ms", lastCause);
389     }

391     private boolean verifyReady() throws IOException, SQLException {
392         // check TCP connection
393         final InetAddress localhost = InetAddress.getLoopbackAddress();
394         try (Socket sock = new Socket()) {
395             sock.setSoTimeout((int) Duration.ofMillis(500).toMillis());
396             sock.connect(new InetSocketAddress(localhost, port), (int) Duration.ofMillis(500).toMillis());
397         } catch (ConnectException e) {
398             return false;
399         }
400 
401         // check JDBC connection
402         try (Connection c = createDefaultDataSource().getConnection();
403                 Statement s = c.createStatement();
404                 ResultSet rs = s.executeQuery("SELECT 1")) {
405             checkState(rs.next(), "expecting single row");
406             checkState(rs.getInt(1) == 1, "expecting 1 as result");
407             checkState(!rs.next(), "expecting single row");
408             return true;
409         }
410     }

426      * Closing an {@link EmbeddedPostgres} instance shuts down the connected database instance.
427      */
428     @Override
429     public void close() throws IOException {
430         if (closed.getAndSet(true)) {
431             return;
432         }
433 
434         try {
435             stopDatabase(this.dataDirectory);
436         } catch (final Exception e) {
437             logger.error("could not stop pg:", e);
438         }
439 
440         unlock();
441 
442         if (removeDataOnShutdown) {
443             try {
444                 EmbeddedUtil.rmdirs(dataDirectory);
445             } catch (Exception e) {
446                 logger.error(format("Could not clean up directory %s:", dataDirectory.getAbsolutePath()), e);
447             }
448         } else {
449             logger.debug(format("preserved data directory %s", dataDirectory.getAbsolutePath()));
450         }
451     }

464     @SuppressFBWarnings("RCN_REDUNDANT_NULLCHECK_OF_NONNULL_VALUE")
465     private void cleanOldDataDirectories(File parentDirectory) {
466         final File[] children = parentDirectory.listFiles();
467         if (children == null) {
468             return;
469         }
470         for (final File dir : children) {
471             if (!dir.isDirectory()) {
472                 continue;
473             }
474 
475             // only ever touch known data directories.
476             if (!dir.getName().startsWith(DATA_DIRECTORY_PREFIX)) {
477                 continue;
478             }
479 
480             // only touch data directories that hold a lock file.
481             final File lockFile = new File(dir, LOCK_FILE_NAME);
482             if (!lockFile.exists()) {
483                 continue;
484             }
485 
486             // file must have a minimum age. This can not be the same check as
487             // the exists b/c non-existant files return 0 (epoch) as lastModified so
488             // they are considered "ancient".
489             if (System.currentTimeMillis() - lockFile.lastModified() < MINIMUM_AGE_IN_MS) {
490                 continue;
491             }
492 
493             try (FileOutputStream fos = new FileOutputStream(lockFile);
494                     FileLock lock = fos.getChannel().tryLock()) {
495                 if (lock != null) {
496                     logger.debug(format("found stale data directory %s", dir));
497                     if (new File(dir, "postmaster.pid").exists()) {
498                         try {
499                             stopDatabase(dir);
500                             logger.debug("shut down orphaned database!");
501                         } catch (Exception e) {
502                             logger.warn(format("failed to orphaned database in %s:", dir), e);
503                         }
504                     }
505                     EmbeddedUtil.rmdirs(dir);
506                 }
507             } catch (final OverlappingFileLockException e) {
508                 // The directory belongs to another instance in this VM.
509                 logger.trace("while cleaning old data directories:", e);
510             } catch (final Exception e) {
511                 logger.warn("while cleaning old data directories:", e);
512             }
513         }
514     }

521     private Process spawn(@Nullable String processName, List<String> commandAndArgs, boolean debug) throws IOException {
522         final ProcessBuilder builder = new ProcessBuilder(commandAndArgs);
523         builder.redirectErrorStream(true);
524         builder.redirectError(errorRedirector);
525         builder.redirectOutput(outputRedirector);
526         final Process process = builder.start();
527 
528         processName = processName != null ? processName : process.info().command().map(EmbeddedUtil::getFileBaseName).orElse("<unknown>");
529         String name = format("%s (%d)", processName, process.pid());
530 
531         ProcessOutputLogger.logOutput(debug, logger, name, process);
532         return process;
533     }

536     private Stopwatch system(List<String> commandAndArgs, boolean debug) throws IOException {
537         checkArgument(commandAndArgs.size() > 0, "No commandAndArgs given!");
538         String prefix = EmbeddedUtil.getFileBaseName(commandAndArgs.get(0));
539 
540         Stopwatch watch = Stopwatch.createStarted();
541         Process process = spawn(prefix, commandAndArgs, debug);
542         try {
543             if (process.waitFor() != 0) {
544                 try (InputStreamReader reader = new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8)) {
545                     throw new IllegalStateException(format("Process %s failed%n%s",
546                             commandAndArgs, CharStreams.toString(reader)));
547                 }
548             }
549         } catch (InterruptedException e) {
550             Thread.currentThread().interrupt();
551         }
552 
553         return watch;
554     }
