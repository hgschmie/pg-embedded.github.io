48  /**
49   * Loads a native binary installation and returns the location of it.
50   */
51  public final class TarXzCompressedBinaryManager implements NativeBinaryManager {
52  
53      private static final Logger LOG = LoggerFactory.getLogger(TarXzCompressedBinaryManager.class);
54  
55      private static final Map<NativeBinaryLocator, File> KNOWN_INSTALLATIONS = new ConcurrentHashMap<>();
56  
57      private final Lock prepareBinariesLock = new ReentrantLock();
58  
59      private final File installationBaseDirectory;
60      private final String lockFileName;
61      private final NativeBinaryLocator nativeBinaryLocator;
62  
63      /**
64       * Creates a new binary manager for tar-xz compressed archives.
65       * <p>
66       * The implementation of {@link NativeBinaryLocator} to locate the stream that gets unpacked must satisfy the following criteria:
67       * <ul>
68       *     <li>It must implement {@link #equals(Object)} and {@link #hashCode()}.</li>
69       *     <li>It should implement {@link #toString()} to return meaningful information about the locator.</li>
70       *     <li>It must allow multiple calls to {@link NativeBinaryLocator#getInputStream()} which all return the same, byte-identical contents.
71       *     The operation should be cheap as it may be called multiple times.</li>
72       * </ul>
73       *
74       * @param installationBaseDirectory Base directory in which the binary distribution is unpacked. Must not be null.
75       * @param lockFileName              Name of a file to use as file lock when unpacking the disttribution.
76       * @param nativeBinaryLocator       An implementation of {@link NativeBinaryLocator} that satisfies the conditions above. Must not be null.
77       */
78      public TarXzCompressedBinaryManager(@NonNull File installationBaseDirectory,
79              @NonNull String lockFileName,
80              @NonNull NativeBinaryLocator nativeBinaryLocator) {
81          this.installationBaseDirectory = checkNotNull(installationBaseDirectory, "installationBaseDirectory is null");
82          this.lockFileName = checkNotNull(lockFileName, "lockFileName is null");
83          this.nativeBinaryLocator = checkNotNull(nativeBinaryLocator, "nativeBinaryLocator is null");
84  
85          checkState(this.installationBaseDirectory.setWritable(true, false),
86                  "Could not make install base directory %s writable!", this.installationBaseDirectory);
87      }
88  
89      @Override
90      @NonNull
91      public File getLocation() throws IOException {
92  
93          // the installation cache saves ~ 1% CPU according to the profiler
94          File installationDirectory = KNOWN_INSTALLATIONS.get(nativeBinaryLocator);
95          if (installationDirectory != null && installationDirectory.exists()) {
96              return installationDirectory;
97          }
98  
99          prepareBinariesLock.lock();
100         try {
101             String installationIdentifier = nativeBinaryLocator.getIdentifier();
102             installationDirectory = new File(installationBaseDirectory, installationIdentifier);
103             EmbeddedUtil.mkdirs(installationDirectory);
104 
105             final File unpackLockFile = new File(installationDirectory, lockFileName);
106             final File installationExistsFile = new File(installationDirectory, ".exists");
107 
108             if (!installationExistsFile.exists()) {
109                 try (FileOutputStream lockStream = new FileOutputStream(unpackLockFile);
110                         FileLock unpackLock = lockStream.getChannel().tryLock()) {
111                     if (unpackLock != null) {
112                         checkState(!installationExistsFile.exists(), "unpack lock acquired but .exists file is present " + installationExistsFile);
113                         LOG.info("extracting archive...");
114                         try (InputStream archiveStream = nativeBinaryLocator.getInputStream()) {
115                             extractTxz(archiveStream, installationDirectory.getPath());
116                             checkState(installationExistsFile.createNewFile(), "couldn't create %s file!", installationExistsFile);
117                         }
118                     } else {
119                         // the other guy is unpacking for us.
120                         int maxAttempts = 60;
121                         while (!installationExistsFile.exists() && --maxAttempts > 0) { // NOPMD
122                             Thread.sleep(1000L);
123                         }
124                         checkState(installationExistsFile.exists(), "Waited 60 seconds for archive to be unpacked but it never finished!");
125                     }
126                 } finally {
127                     if (unpackLockFile.exists() && !unpackLockFile.delete()) {
128                         LOG.error(format("could not remove lock file %s", unpackLockFile.getAbsolutePath()));
129                     }
130                 }
131             }
132 
133             KNOWN_INSTALLATIONS.putIfAbsent(nativeBinaryLocator, installationDirectory);
134             LOG.debug(format("Unpacked archive at %s", installationDirectory));
135             return installationDirectory;
136 
137         } catch (final InterruptedException e) {
138             Thread.currentThread().interrupt();
139             throw new IOException(e);
140         } finally {
141             prepareBinariesLock.lock();
142         }
143     }
144 
145     /**
146      * Unpack archive compressed by tar with xz compression.
147      *
148      * @param stream    A tar-xz compressed data stream.
149      * @param targetDir The directory to extract the content to.
150      */
151     private static void extractTxz(InputStream stream, String targetDir) throws IOException {
152         try (XZInputStream xzIn = new XZInputStream(stream);
153                 TarArchiveInputStream tarIn = new TarArchiveInputStream(xzIn)) {
154             final Phaser phaser = new Phaser(1);
155             TarArchiveEntry entry;
156 
157             while ((entry = tarIn.getNextTarEntry()) != null) { //NOPMD
158                 final String individualFile = entry.getName();
159                 final File fsObject = new File(targetDir, individualFile);
160                 final Path fsPath = fsObject.toPath();
161                 if (Files.exists(fsPath, LinkOption.NOFOLLOW_LINKS) && !Files.isDirectory(fsPath, LinkOption.NOFOLLOW_LINKS)) {
162                     Files.delete(fsPath);
163                     LOG.debug(format("Deleting existing entry %s", fsPath));
164                 }
165 
166                 if (entry.isSymbolicLink() || entry.isLink()) {
167                     Path target = FileSystems.getDefault().getPath(entry.getLinkName());
168                     Files.createSymbolicLink(fsPath, target);
169                 } else if (entry.isFile()) {
170                     byte[] content = new byte[(int) entry.getSize()];
171                     int read = tarIn.read(content, 0, content.length);
172                     checkState(read != -1, "could not read %s", individualFile);
173                     EmbeddedUtil.mkdirs(fsObject.getParentFile());
174 
175                     final AsynchronousFileChannel fileChannel = AsynchronousFileChannel.open(fsPath, CREATE, WRITE); //NOPMD
176                     final ByteBuffer buffer = ByteBuffer.wrap(content); //NOPMD
177 
178                     phaser.register();
179                     fileChannel.write(buffer, 0, fileChannel, new CompletionHandler<Integer, Channel>() {
180                         @Override
181                         public void completed(Integer written, Channel channel) {
182                             closeChannel(channel);
183                         }
184 
185                         @Override
186                         public void failed(Throwable error, Channel channel) {
187                             LOG.error(format("could not write file %s", fsObject.getAbsolutePath()), error);
188                             closeChannel(channel);
189                         }
190 
191                         private void closeChannel(Channel channel) {
192                             try {
193                                 channel.close();
194                             } catch (IOException e) {
195                                 LOG.error("While closing channel:", e);
196                             } finally {
197                                 phaser.arriveAndDeregister();
198                             }
199                         }
200                     });
201                 } else if (entry.isDirectory()) {
202                     EmbeddedUtil.mkdirs(fsObject);
203                 } else {
204                     throw new IOException(format("Unsupported entry in tar file found: %s", individualFile));
205                 }
206 
207                 if (individualFile.startsWith("bin/") || individualFile.startsWith("./bin/")) {
208                     if (!fsObject.setExecutable(true, false)) {
209                         throw new IOException(format("Could not make %s executable!", individualFile));
210                     }
211                 }
212             }
213 
214             phaser.arriveAndAwaitAdvance();
215         }
216     }
